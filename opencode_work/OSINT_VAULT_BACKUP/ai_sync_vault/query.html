HUD PIT · NPI (Non‑Profiteers) · Environmental ESA — cross‑referenced by state 
-- MULTI-STREAM CORRELATION EXECUTION
-- CROSS-REFERENCING HUD PIT MATRIX, NPI DELTAS, AND ENVIRONMENTAL ESAs

WITH PIT_Metrics AS (
  -- Stream 2: Aggregate HUD Point-in-Time Counts per State
  SELECT
    state,
    SUM(pit.total_homeless) AS total_homeless_population,
    SUM(pit.sheltered_homeless) AS total_sheltered,
    SUM(pit.unsheltered_homeless) AS total_unsheltered,
    MAX(pit.count_year) AS latest_pit_year
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(hud_pit_list) AS pit
  GROUP BY state
),

NPI_Metrics AS (
  -- Stream 4: Flatten corporate leakage parameters per State
  SELECT
    state,
    COUNT(npi.npi_id) AS total_active_shell_entities,
    SUM(npi.unaccounted_fund_delta) AS aggregate_unaccounted_leakage
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(non_profiteers_index) AS npi
  GROUP BY state
),

ESA_Metrics AS (
  -- Stream 5: Isolate toxic footprints per State
  SELECT
    state,
    COUNT(esa.site_id) AS total_monitored_toxic_sites,
    COUNTIF(esa.test_multiplier >= 10.0) AS severe_chemical_safety_breaches,
    MAX(CASE WHEN esa.site_id = 'ENV-HBNC-001' THEN esa.test_multiplier ELSE NULL END) AS target_hbnc_crvi_multiplier
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(environmental_site_assessments) AS esa
  GROUP BY state
)

SELECT
  base.state,
  base.state_name,
  pit.latest_pit_year,
  pit.total_homeless_population,
  pit.total_unsheltered,
  npi.total_active_shell_entities,
  npi.aggregate_unaccounted_leakage,
  esa.total_monitored_toxic_sites,
  esa.severe_chemical_safety_breaches,
  esa.target_hbnc_crvi_multiplier AS hbnc_toxic_excess_rate
FROM `noble-beanbag-497411-m4.national_audits.all_state_records` AS base
LEFT JOIN PIT_Metrics AS pit ON base.state = pit.state
LEFT JOIN NPI_Metrics AS npi ON base.state = npi.state
LEFT JOIN ESA_Metrics AS esa ON base.state = esa.state
ORDER BY npi.aggregate_unaccounted_leakage DESC;
