### III. GOOGLESQL FORENSIC EXTRACTION QUERY
To map these relationships without creating corruptive Cartesian products (strictly adhering to the **Single Array Rule**), the following query isolates and correlates the state audits, CoC funding records, and the Non-Profiteers Index using CTEs joined on the `state` anchor.

```sql
WITH state_audit_parameters AS (
  -- Extract CA Audit 2023-102.1 parameters to establish the macro blind spot
  SELECT 
    state,
    audit.report_num,
    audit.report_title,
    audit.taxpayer_funds_reviewed AS state_untracked_footprint
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(performance_audit_list) AS audit
  WHERE state = 'CA' 
    AND audit.report_num = '2023-102.1'
),

mercy_house_target AS (
  -- Isolate Mercy House network footprints from the Non-Profiteers Index
  SELECT 
    state,
    npi.organization_name,
    npi.opencorporates_url,
    npi.cms_billing_code,
    npi.unaccounted_fund_delta
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(non_profiteers_index) AS npi
  WHERE state = 'CA' 
    AND UPPER(npi.organization_name) LIKE '%MERCY HOUSE%'
),

coc_funding_footprint AS (
  -- Aggregate active CoC Funding parameters for California
  SELECT 
    state,
    SUM(coc.award_amount) AS total_coc_award_pool
  FROM `noble-beanbag-497411-m4.national_audits.all_state_records`,
  UNNEST(coc_continuum_list) AS coc
  WHERE state = 'CA'
  GROUP BY state
)

SELECT 
  sa.state,
  sa.report_num AS audit_reference,
  sa.state_untracked_footprint AS macro_audit_untracked_pool,
  mh.organization_name AS target_entity,
  mh.opencorporates_url,
  cf.total_coc_award_pool,
  mh.unaccounted_fund_delta AS baseline_target_delta,
  -- Apply the registered Mathematical Ingestion UDF to calculate real leakage
  `noble-beanbag-497411-m4.national_audits.calculate_fund_leakage`(
    cf.total_coc_award_pool, 
    (cf.total_coc_award_pool - mh.unaccounted_fund_delta)
  ) AS verified_leakage_delta
FROM state_audit_parameters sa
JOIN mercy_house_target mh ON sa.state = mh.state
JOIN coc_funding_footprint cf ON sa.state = cf.state;
```
